7804  static basic_json diff(const basic_json& source, const basic_json& target,
7805  const std::string& path = "")
7806  {
7807  // the patch
7808  basic_json result(value_t::array);
7809 
7810  // if the values are the same, return empty patch
7811  if (source == target)
7812  {
7813  return result;
7814  }
7815 
7816  if (source.type() != target.type())
7817  {
7818  // different types: replace value
7819  result.push_back(
7820  {
7821  {"op", "replace"}, {"path", path}, {"value", target}
7822  });
7823  return result;
7824  }
7825 
7826  switch (source.type())
7827  {
7828  case value_t::array:
7829  {
7830  // first pass: traverse common elements
7831  std::size_t i = 0;
7832  while (i < source.size() and i < target.size())
7833  {

7836  result.insert(result.end(), temp_diff.begin(), temp_diff.end());
7837  ++i;
7838  }

7841  // in a second pass, traverse the remaining elements
7842

7845  while (i < source.size())
7846  {
7847  // add operations in reverse order to avoid invalid
7848  // indices

7850  {
7851  {"op", "remove"},
7852  {"path", path + "/" + std::to_string(i)}
7853  }));
7854  ++i;
7855  }
7856 
7857  // add other remaining elements
7858  while (i < target.size())
7859  {
7860  result.push_back(
7861  {
7862  {"op", "add"},
7863  {"path", path + "/" + std::to_string(i)},
7864  {"value", target[i]}
7865  });
7866  ++i;
7867  }
7868 
7869  break;
7870  }
7871 
7872  case value_t::object:
7873  {
7874  // first pass: traverse this object's elements
7875  for (auto it = source.cbegin(); it != source.cend(); ++it)
7876  {
7877  // escape the key name to be used in a JSON patch
7878  const auto key = json_pointer::escape(it.key());
7879

7881  {
7882  // recursive call to compare object values at key it

7884  result.insert(result.end(), temp_diff.begin(), temp_diff.end());
7885  }
7886  else
7887  {
